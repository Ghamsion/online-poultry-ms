    "## Step 3 \u2014 Verify the dataset and class balance\n",
    "\n",
    "This reproduces a real version of Table 3.2 from Chapter 3 \u2014 actual box\n",
    "counts per class and per split, not the placeholder numbers.\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},
   "execution_count": null,
   "outputs": [],
   "source": [
    "import os, glob\n",
    "from collections import Counter\n",
    "\n",
    "CLASS_NAMES = {0: \"live\", 1: \"dead\", 2: \"egg\"}\n",
    "\n",
    "def count_split(split_dir):\n",
    "    label_files = glob.glob(os.path.join(split_dir, \"labels\", \"*.txt\"))\n",
    "    counts = Counter()\n",
    "    n_images = len(glob.glob(os.path.join(split_dir, \"images\", \"*\")))\n",
    "    for lf in label_files:\n",
    "        with open(lf) as f:\n",
    "            for line in f:\n",
    "                line = line.strip()\n",
    "                if not line:\n",
    "                    continue\n",
    "                cls_id = int(line.split()[0])\n",
    "                counts[CLASS_NAMES.get(cls_id, cls_id)] += 1\n",
    "    return n_images, counts\n",
    "\n",
    "print(f\"{'Split':<10}{'Images':<10}{'live':<8}{'dead':<8}{'egg':<8}\")\n",
    "for split in [\"train\", \"val\", \"test\"]:\n",
    "    split_path = os.path.join(DATASET_DIR, split)\n",
    "    if not os.path.isdir(split_path):\n",
    "        print(f\"  (no '{split}' folder found at {split_path} \u2014 check your dataset structure)\")\n",
    "        continue\n",
    "    n_img, counts = count_split(split_path)\n",
    "    print(f\"{split:<10}{n_img:<10}{counts.get('live',0):<8}{counts.get('dead',0):<8}{counts.get('egg',0):<8}\")\n",
    "\n",
    "print()\n",
    "print(\"Copy this real table into Table 3.2 of Chapter 3, replacing the placeholder numbers.\")\n"

    "## Step 4 (optional) \u2014 Oversample the `dead` class\n",
    "\n",
    "Ultralytics' YOLOv8 trainer does not expose a simple per-class loss-weight\n",
    "argument the way Chapter 5's Table 5.2 currently describes (\"class loss\n",
    "weight 3.0 for dead\"). The practical, honest way to fight class imbalance is\n",
    "**oversampling**: duplicate images that contain a `dead` box a few times in\n",
    "the training split so the model sees them more often per epoch.\n",
    "\n",
    "If you do this, update Table 5.2 in Chapter 5 to say \"stratified oversampling\n",
    "of the dead class (\u00d7N duplication)\" instead of \"class loss weight 3.0/1.0\" \u2014\n",
    "say what you actually did.\n",
    "\n",
    "Skip this cell if your `dead` class already has a reasonable number of images\n",
    "relative to `live`.\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},
   "execution_count": null,
   "outputs": [],
   "source": [
    "import shutil, os, glob\n",
    "\n",
    "OVERSAMPLE_FACTOR = 3  # duplicate each dead-containing image this many extra times\n",
    "\n",
    "train_img_dir = os.path.join(DATASET_DIR, \"train\", \"images\")\n",
    "train_lbl_dir = os.path.join(DATASET_DIR, \"train\", \"labels\")\n",
    "\n",
    "dead_label_files = []\n",
    "for lf in glob.glob(os.path.join(train_lbl_dir, \"*.txt\")):\n",
    "    with open(lf) as f:\n",
    "        if any(line.strip().startswith(\"1 \") for line in f):\n",
    "            dead_label_files.append(lf)\n",
    "\n",
    "print(f\"Found {len(dead_label_files)} training images containing a 'dead' box.\")\n",
    "\n",
    "for lf in dead_label_files:\n",
    "    base = os.path.splitext(os.path.basename(lf))[0]\n",
    "    img_candidates = glob.glob(os.path.join(train_img_dir, base + \".*\"))\n",
    "    if not img_candidates:\n",
    "        continue\n",
    "    img_path = img_candidates[0]\n",
    "    ext = os.path.splitext(img_path)[1]\n",
    "    for i in range(OVERSAMPLE_FACTOR):\n",
    "        new_base = f\"{base}_dup{i}\"\n",
    "        shutil.copy(img_path, os.path.join(train_img_dir, new_base + ext))\n",
    "        shutil.copy(lf, os.path.join(train_lbl_dir, new_base + \".txt\"))\n",
    "\n",
    "print(\"Oversampling complete. Re-run the Step 3 cell to see the updated class balance.\")\n"
   ]
